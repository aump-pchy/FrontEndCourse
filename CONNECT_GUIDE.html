Login · Auth Store
คนที่ 1

แก้ไฟล์ 2 ไฟล์ — auth store และ LoginView

แก้ src/stores/auth.js

stores/auth.js
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import api from '@/api'

export const useAuthStore = defineStore('auth', () => {
  const user  = ref(JSON.parse(localStorage.getItem('user') || 'null'))
  const token = ref(localStorage.getItem('token') || '')

  const isLoggedIn = computed(() => !!token.value)
  const isAdmin    = computed(() => user.value?.role === 'admin')

  async function login(email, password) {
    const { data } = await api.post('/auth/login', { email, password })
    token.value = data.token
    user.value  = data.user
    localStorage.setItem('token', data.token)
    localStorage.setItem('user', JSON.stringify(data.user))
  }

  function logout() {
    token.value = ''
    user.value  = null
    localStorage.removeItem('token')
    localStorage.removeItem('user')
  }

  return { user, token, isLoggedIn, isAdmin, login, logout }
})

แก้ src/views/LoginView.vue — ส่วน script

LoginView.vue <script setup>
import { ref } from 'vue'
import { useRouter } from 'vue-router'
import { useAuthStore } from '@/stores/auth'

const auth     = useAuthStore()
const router   = useRouter()
const email    = ref('')
const password = ref('')
const error    = ref('')
const loading  = ref(false)

async function handleLogin() {
  error.value   = ''
  loading.value = true
  try {
    await auth.login(email.value, password.value)
    router.push('/memos') // ← login สำเร็จ ไปหน้า memos
  } catch (err) {
    error.value = 'อีเมลหรือรหัสผ่านไม่ถูกต้อง'
  } finally {
    loading.value = false
  }
}

MemoList · ดึงข้อมูลมาแสดง
คนที่ 2

แก้ src/stores/memo.js

stores/memo.js
import { defineStore } from 'pinia'
import { ref } from 'vue'
import api from '@/api'

export const useMemoStore = defineStore('memo', () => {
  const memos   = ref([])
  const current = ref(null)
  const loading = ref(false)

  async function fetchMemos(params = {}) {
    loading.value = true
    try {
      const { data } = await api.get('/memos', { params })
      memos.value = data.data
    } finally { loading.value = false }
  }

  async function fetchMemo(id) {
    const { data } = await api.get(`/memos/${id}`)
    current.value = data.data
  }

  async function createMemo(payload) {
    const { data } = await api.post('/memos', payload)
    return data
  }

  async function updateMemo(id, payload) {
    const { data } = await api.put(`/memos/${id}`, payload)
    return data
  }

  async function deleteMemo(id) {
    await api.delete(`/memos/${id}`)
    memos.value = memos.value.filter(m => m.id !== id)
  }

  return { memos, current, loading, fetchMemos, fetchMemo, createMemo, updateMemo, deleteMemo }
})

MemoForm · สร้าง / แก้ไข
คนที่ 3

⚡ MemoForm ใช้หน้าเดียวกันทั้ง Create และ Edit — เช็คจาก route.params.id ว่ามีหรือเปล่า

MemoFormView.vue <script setup>
import { ref, onMounted, computed } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useMemoStore } from '@/stores/memo'

const store  = useMemoStore()
const route  = useRoute()
const router = useRouter()

// เช็คว่าเป็นหน้า Edit หรือ Create
const isEdit = computed(() => !!route.params.id)

const form = ref({
  subject: '', recipient: '', operator: '',
  content: '', memo_number: ''
})
const error = ref('')

// ถ้าเป็น Edit → โหลดข้อมูลเดิมมาใส่ฟอร์ม
onMounted(async () => {
  if (isEdit.value) {
    await store.fetchMemo(route.params.id)
    form.value = { ...store.current }
  }
})

async function handleSubmit() {
  error.value = ''
  try {
    if (isEdit.value) {
      await store.updateMemo(route.params.id, form.value)
    } else {
      await store.createMemo(form.value)
    }
    router.push('/memos')
  } catch (err) {
    error.value = err.response?.data?.message || 'เกิดข้อผิดพลาด'
  }
}

MemoDetailView.vue <script setup>
import { onMounted, computed } from 'vue'
import { useRoute } from 'vue-router'
import { useMemoStore } from '@/stores/memo'

const store = useMemoStore()
const route = useRoute()
const memo  = computed(() => store.current)

// โหลดข้อมูลตอนเปิดหน้า
onMounted(() => store.fetchMemo(route.params.id))

เปิด Route Guard · ทำหลังสุด
👥 ทุกคน
⚠️ ทำขั้นนี้ หลังจากทุก feature ทดสอบผ่านแล้วเท่านั้น

1. เปิด verifyToken ใน backend routes ทุกไฟล์

routes/memos.js และ routes/users.js
// เอา comment ออก
router.use(verifyToken)
router.use(verifyToken, verifyAdmin)

2. เปิด Route Guard ใน src/router/index.js

router/index.js
router.beforeEach((to, _, next) => {
  const auth = useAuthStore()
  if (!to.meta.public && !auth.isLoggedIn) return next('/login')
  if (to.name === 'login' && auth.isLoggedIn) return next('/memos')
  next()
})
